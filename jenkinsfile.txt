pipeline{
  agent any
  stages{
    stage('checkout'){
      steps{
        echo 'Checking out Repo'
        git branch : 'main', url : 'https://github.com/arkadealpha1/DevOpsRepo.git'
      }
    }
    stage('Publish'){
      steps{
        publishHTML([
          allowmissing:true,
          alwaysLinktoLastBuild:false,
          KeepAll:false,
          reportDir:'.',
          reportFiles:'firstfirst.html',
          reportName:'MY HTML PIPE PAGE'
          ])
      }
    }
  }

}
